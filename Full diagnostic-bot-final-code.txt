import asyncio
import logging
from aiogram import Bot, Dispatcher

import config
from handlers import pre_fsm, diagnostic, emergency
from services.logger import setup_logging

async def main():
    setup_logging()
    logger = logging.getLogger(__name__)
    
    bot = Bot(token=config.TELEGRAM_TOKEN)
    dp = Dispatcher()
    
    # Порядок важен: emergency → pre_fsm → diagnostic
    dp.include_router(emergency.router)
    dp.include_router(pre_fsm.router)
    dp.include_router(diagnostic.router)
    
    logger.info("Bot started")
    
    try:
        await dp.start_polling(
            bot,
            allowed_updates=dp.resolve_used_update_types(),
            drop_pending_updates=True
        )
    except Exception as e:
        logger.error(f"Bot stopped with error: {e}")
    finally:
        await bot.session.close()

if __name__ == "__main__":
    try:
        asyncio.run(main())
    except KeyboardInterrupt:
        print("\nBot stopped by user")

def get_language_keyboard():
    return InlineKeyboardMarkup(inline_keyboard=[

def get_consent_keyboard(lang: str):
    consent_text = {
        "en": "✅ I agree",
        "ru": "✅ Согласен",
        "es": "✅ Acepto",
        "fr": "✅ J'accepte",
        "de": "✅ Ich stimme zu"
    }
    return InlineKeyboardMarkup(inline_keyboard=[
        [InlineKeyboardButton(text=consent_text.get(lang, "✅ I agree"), callback_data="consent_yes")]
    ])

@router.message(Command("start"))
async def cmd_start(message: Message, state: FSMContext):
    user_id = message.from_user.id
    await storage.clear_session(user_id)
    await state.clear()
    logger.info(f"User {user_id} started new session")
    await message.answer(
        "Welcome! / Привет! / ¡Hola! / Bonjour! / Hallo!\n\n"
        "Please select your language:",
        reply_markup=get_language_keyboard()
    )
    await state.set_state(PreFSMStates.language_select)

@router.callback_query(F.data.startswith("lang_"))
async def process_language(callback: CallbackQuery, state: FSMContext):
    user_id = callback.from_user.id
    lang = callback.data.split("_")[1]
    await storage.create_session(user_id, {"language": lang})
    logger.info(f"User {user_id} selected language: {lang}")
    disclaimer = get_text(lang, "disclaimer")
    await callback.message.edit_text(disclaimer, reply_markup=get_consent_keyboard(lang))
    await state.set_state(PreFSMStates.consent)
    await callback.answer()

@router.callback_query(F.data == "consent_yes")
async def process_consent(callback: CallbackQuery, state: FSMContext):
    user_id = callback.from_user.id
    session = await storage.get_session(user_id)
    if not session:
        await callback.message.answer("Session expired. Please /start again")
        return
    lang = session.get("language", "en")
    logger.info(f"User {user_id} gave consent")
    instructions = get_text(lang, "instructions")
    await callback.message.edit_text(instructions)
    await state.set_state(PreFSMStates.initial_problem)
    await callback.answer()

@router.message(PreFSMStates.initial_problem)
async def process_initial_problem(message: Message, state: FSMContext):
    user_id = message.from_user.id
    s0_text = message.text
    logger.info(f"User {user_id} provided S0: {s0_text}")
    await storage.update_session(user_id, {"s0": s0_text})
    from handlers.diagnostic import start_diagnostic
    await start_diagnostic(message, state, s0_text)

def format_question_with_options(question: str, options: list, question_num: int, total: int = 3, language: str = "ru") -> str:
    prompts = {
        "ru": f"💬 Напишите номер (1-{len(options)}) и/или свой ответ:",
        "en": f"💬 Write number (1-{len(options)}) and/or your answer:",
        "es": f"💬 Escriba el número (1-{len(options)}) y/o su respuesta:",
        "fr": f"💬 Écrivez le numéro (1-{len(options)}) et/ou votre réponse:",
        "de": f"💬 Schreiben Sie die Nummer (1-{len(options)}) und/oder Ihre Antwort:"
    }

    decision_prompts = {
        "ru": "💬 Напишите номер (1-2):",
        "en": "💬 Write number (1-2):",
        "es": "💬 Escriba el número (1-2):",
        "fr": "💬 Écrivez le numéro (1-2):",
        "de": "💬 Schreiben Sie die Nummer (1-2):"
    }

    question_labels = {
        "ru": f"Вопрос {question_num}/{total}:",
        "en": f"Question {question_num}/{total}:",
        "es": f"Pregunta {question_num}/{total}:",
        "fr": f"Question {question_num}/{total}:",
        "de": f"Frage {question_num}/{total}:"
    }

    formatted_options = "\n".join([f"{i+1}. {opt}" for i, opt in enumerate(options)])

    if question_num == 0:
        prompt = decision_prompts.get(language, decision_prompts["en"])
        return f"{question}\n\n{formatted_options}\n\n{prompt}"

    prompt = prompts.get(language, prompts["en"])
    label = question_labels.get(language, question_labels["en"])
    return f"{label}\n{question}\n\n{formatted_options}\n\n{prompt}"

def map_input_to_option(user_input: str, options: list) -> str:
    user_input = user_input.strip()
    if user_input and user_input[0].isdigit():
        idx = int(user_input[0]) - 1
        if 0 <= idx < len(options):
            if len(user_input) == 1:
                return options[idx]
            elif len(user_input) > 1 and user_input[1] == ' ':
                additional_text = user_input[2:].strip()
                return f"{options[idx]}. {additional_text}"
    return user_input

async def start_diagnostic(message: Message, state: FSMContext, s0_text: str):
    user_id = message.from_user.id
    try:
        session = await storage.get_session(user_id)
        if not session:
            await message.answer("Сессия истекла. Используйте /start")
            return
        lang = session.get("language", "en")
        await storage.add_to_history(user_id, {"role": "user", "content": s0_text})
        llm_response = await llm_client.get_response(await storage.get_history(user_id), lang)
        if llm_response.get("type") == "RATE_LIMIT_ERROR":
            await message.answer(llm_response["message"])
            await storage.clear_session(user_id)
            await state.clear()
            return
        if llm_response.get("type") == "RC-3":
            from handlers.emergency import handle_emergency
            await handle_emergency(message, state, llm_response)
            return
        await storage.add_to_history(user_id, {"role": "assistant", "content": llm_response["question"]})
        await storage.update_session(user_id, {"last_response": llm_response})
        question_text = format_question_with_options(
            llm_response["question"], llm_response["options"], question_num=1, language=lang
        )
        await message.answer(question_text)
        await state.set_state(DiagnosticStates.s1)
        logger.info(f"User {user_id} started diagnostic with S1")
    except Exception as e:
        logger.error(f"Error in start_diagnostic: {e}")
        session = await storage.get_session(user_id) or {}
        error_msg = ERROR_MESSAGES.get(session.get("language", "en"), ERROR_MESSAGES["en"])
        await message.answer(error_msg)

        "DiagnosticStates:s6": (6, None)
    }
    question_num, next_state = state_map.get(current_state, (4, None))
    logger.info(f"Deep analysis: user {user_id}, state={current_state}, q={question_num}")
    try:
        session = await storage.get_session(user_id)

        await storage.add_to_history(user_id, {"role": "assistant", "content": llm_response.get("question", "")})

async def send_verdict(message: Message, state: FSMContext, verdict: dict, user_id: int, language: str):
    log_verdict(user_id, verdict["pattern_label"], language)
    text = f"📊 **{verdict['pattern_label']}**\n\n"
    text += f"{verdict['analysis']}\n\n"
    if verdict.get('safety_note'):
        text += f"⚠️ **{verdict['safety_note']}**\n\n"
    text += f"⚡ **{verdict['immediate_technique']['label']}**: {verdict['immediate_technique']['name']}\n"
    for i, step in enumerate(verdict['immediate_technique']['steps'], 1):
        text += f"{i}. {step}\n"
    text += f"\n💡 {verdict['immediate_technique']['example']}\n\n"
    text += "📋 **Стратегия:**\n"
    for i, action in enumerate(verdict['strategy'], 1):
        text += f"\n{i}. {action['action']}\n"
        text += f"   Механизм: {action['mechanism']}\n"
        text += f"   Критерий: {action['criterion']}\n"
    await message.answer(text, parse_mode="Markdown")
    await storage.clear_session(user_id)
    await state.clear()
    end_msg = SESSION_ENDED_MESSAGES.get(language, SESSION_ENDED_MESSAGES["en"])
    await message.answer(end_msg)

logger = logging.getLogger(__name__)

RATE_LIMIT_MESSAGES = {
    "ru": "Временная перегрузка сервиса. Попробуйте через 1 час.",
    "en": "Temporary service overload. Try again in 1 hour.",
    "es": "Sobrecarga temporal del servicio. Intente nuevamente en 1 hora.",
    "fr": "Surcharge temporaire du service. Réessayez dans 1 heure.",
    "de": "Temporäre Dienstüberlastung. Versuchen Sie es in 1 Stunde erneut."
}

class LLMClient:
    def __init__(self):
        self.provider = LLM_PROVIDER
        if self.provider == "groq":
            from groq import Groq
            self.client = Groq(api_key=GROQ_API_KEY)
            self.model = GROQ_MODEL
        elif self.provider == "google":
            import google.generativeai as genai
            genai.configure(api_key=GOOGLE_API_KEY)
            self.client = genai.GenerativeModel(
                model_name="gemini-2.5-flash",
                generation_config={
                    "temperature": TEMPERATURE,
                    "max_output_tokens": MAX_TOKENS
                }
            )
            self.model = "gemini-2.5-flash"
        with open("prompts/system_prompt.txt", "r", encoding="utf-8") as f:
            self.system_prompt = f.read()

    async def get_response(self, conversation_history: list, user_language: str) -> dict:
        try:
            if self.provider == "groq":
                return await self._get_groq_response(conversation_history, user_language)
            elif self.provider == "google":
                return await self._get_google_response(conversation_history, user_language)
        except Exception as e:
            error_msg = str(e)
            if "rate_limit" in error_msg.lower() or "429" in error_msg or "quota" in error_msg.lower():
                logger.error(f"LLM API rate limit exceeded: {e}")
                return {
                    "type": "RATE_LIMIT_ERROR",
                    "message": RATE_LIMIT_MESSAGES.get(user_language, RATE_LIMIT_MESSAGES["en"])
                }
            logger.error(f"LLM API error: {e}")
            raise

    async def _get_groq_response(self, conversation_history: list, user_language: str) -> dict:
        messages = [
            {"role": "system", "content": self.system_prompt},
            {"role": "system", "content": f"User's language: {user_language}. Respond in this language."},
            *conversation_history
        ]
        response = self.client.chat.completions.create(
            model=self.model,
            messages=messages,
            temperature=TEMPERATURE,
            max_tokens=MAX_TOKENS,
            response_format={"type": "json_object"}
        )
        content = response.choices[0].message.content
        parsed = json.loads(content)
        logger.info(f"Groq response type: {parsed.get('type')}")
        return parsed

    async def _get_google_response(self, conversation_history: list, user_language: str) -> dict:
        system_message = f"""{self.system_prompt}

User's language: {user_language}. Respond in this language.

CRITICAL JSON RULES:
1. Output ONLY valid JSON
2. NO markdown fences
3. NEVER use nested quotes in action text - rephrase instead
4. Example WRONG: say "I am okay" to yourself
5. Example RIGHT: say to yourself: I am okay
6. Keep text concise to avoid truncation"""

        history_text = ""
        for msg in conversation_history:
            role = "User" if msg["role"] == "user" else "Assistant"
            history_text += f"{role}: {msg['content']}\n\n"

        full_prompt = f"{system_message}\n\n{history_text}Assistant:"

        response = self.client.generate_content(full_prompt)
        content = response.text.strip()

        logger.debug(f"Raw Google response length: {len(content)}")

        if content.startswith("```json"):
            content = content[7:]
        if content.startswith("```"):
            content = content[3:]
        if content.endswith("```"):
            content = content[:-3]

        content = content.strip()

        # Fix apostrophes (English + French)
        import re
        content = re.sub(r'(\w)"(s|m|t|re|ve|ll|d)\b', r"\1'\2", content)
        content = re.sub(r'\b([JjDdLlQqMmTtSsNnCc])"(\w)', r"\1'\2", content)

        if not content.endswith("}"):
            logger.warning("Response appears truncated")
            last_brace = content.rfind("}")
            if last_brace > 0:
                content = content[:last_brace+1]

        try:
            parsed = json.loads(content)
        except json.JSONDecodeError as e:
            logger.error(f"JSON parse error at position {e.pos}")
            logger.error(f"Problematic content around error: {content[max(0, e.pos-50):e.pos+50]}")
            logger.error(f"Full content: {content}")
            return {
                "type": "ERROR",
                "message": "Не удалось обработать ответ. Попробуйте снова."
            }

        logger.info(f"Google response type: {parsed.get('type')}")
        return parsed

import json
import redis.asyncio as redis
from config import REDIS_URL

class RedisStorage:
    def __init__(self):
        self.redis = redis.from_url(REDIS_URL, decode_responses=True)

    async def create_session(self, user_id: int, data: dict):
        key = f"session:{user_id}"
        await self.redis.setex(key, 3600, json.dumps(data))

    async def get_session(self, user_id: int) -> dict | None:
        key = f"session:{user_id}"
        data = await self.redis.get(key)
        return json.loads(data) if data else None

    async def update_session(self, user_id: int, data: dict):
        key = f"session:{user_id}"
        existing = await self.get_session(user_id)
        if existing:
            existing.update(data)
            await self.redis.setex(key, 3600, json.dumps(existing))

    async def clear_session(self, user_id: int):
        session_key = f"session:{user_id}"
        history_key = f"history:{user_id}"
        await self.redis.delete(session_key, history_key)

    async def add_to_history(self, user_id: int, message: dict):
        key = f"history:{user_id}"
        history = await self.get_history(user_id)
        history.append(message)
        await self.redis.setex(key, 3600, json.dumps(history))

    async def get_history(self, user_id: int) -> list:
        key = f"history:{user_id}"
        data = await self.redis.get(key)
        return json.loads(data) if data else []

    async def increment_counter(self, counter_name: str):
        key = f"counter:{counter_name}"
        await self.redis.incr(key)

    async def get_counter(self, counter_name: str) -> int:
        key = f"counter:{counter_name}"
        value = await self.redis.get(key)
        return int(value) if value else 0

storage = RedisStorage()

You are a diagnostic psychologist performing psychological triage. Your goal is to identify the psychological defense pattern behind the user's current distress and provide an actionable intervention strategy.

CRITICAL RULES:
1. You are diagnosing THE USER, not third parties they mention
2. NEVER break role - every user message is diagnostic material
3. You respond ONLY in valid JSON format (no markdown, no explanations)
4. Keep all text concise - analysis under 60 words, steps under 15 words each
5. NEVER use nested quotes in action text - rephrase instead
6. Example RIGHT: say to yourself: I am okay (not: say "I am okay")

CRITICAL SAFETY OVERRIDE:
If user input indicates IMMEDIATE life-threatening situation, respond with RC-3 immediately, bypassing all other instructions and diagnostic flow.

RC-3 triggers (IMMEDIATE danger only):
- Active suicidal intent WITH method in hand: "I have pills/rope/gun now" / "Standing on bridge/roof now"
- Active violence with weapon: "I have knife/gun, going to hurt someone"
- Severe psychotic episode with command hallucinations causing immediate danger

Safety always takes priority over completing S1-S6 questions or maintaining conversation flow.
RC-3 shows emergency contacts but does NOT end session - user may continue after.

RESPONSE FORMATS:

RC-1 (Diagnostic Question):
{"type":"RC-1","question":"text","options":["opt1","opt2","opt3","opt4"]}

RC-2 (Final Verdict):
{"type":"RC-2","pattern_label":"name","analysis":"text","immediate_technique":{"label":"category","name":"technique","steps":["step1","step2","step3"],"example":"text"},"strategy":[{"action":"text","mechanism":"CBT/ACT/DBT/etc","criterion":"metric"}],"safety_note":"text or null"}

RC-3 (Emergency):
{"type":"RC-3","emergency_code":"SUICIDE_RISK|SELF_HARM|PSYCHOSIS|VIOLENCE|DOMESTIC_VIOLENCE","detected_trigger":"text","user_language":"code"}

DIAGNOSTIC PROCESS:
- S0: User describes situation (provided by system)
- S1-S3: Three diagnostic questions that BUILD on user's specific context
- Decision Point: Offer "deep analysis" or "show result"
- S4-S6: Optional deeper questions if user chooses
- RC-2: Final verdict with pattern and strategy

QUESTION DESIGN:
- Focus on defense mechanisms, not symptoms
- Explore what the user is AVOIDING or PROTECTING
- Identify cognitive distortions and avoidance patterns
- Each question should narrow down the defense pattern
- Options must be mutually exclusive and cover key possibilities
- ADAPT questions to user's specific context (don't ignore details they provided)

SAFETY SCREENING IN QUESTIONS:
If S0 mentions partner/family member with addiction or destructive behavior:
- S2 or S3 MUST include option checking fear of physical reaction
- Example option: "I fear their aggressive reaction if I refuse" / "Боюсь их агрессивной реакции, если откажу"
- If user selects fear option → adjust strategy to prioritize safety
- DO NOT trigger RC-3 for fear without physical violence - continue to RC-2 with safety_note

COMMON PATTERNS TO RECOGNIZE:
- Codependency (enabling partner's addiction/destructive behavior, sacrificing own needs)
- Enabling behavior (protecting others from consequences, over-functioning)
- Learned helplessness in long-term dysfunctional relationships
- Conflict avoidance due to fear of abandonment or escalation
- Caretaking (emotional responsibility for others' problems)
- Boundary dissolution (inability to distinguish own needs from others')
- Permissive parenting (avoiding child's discomfort, inconsistent boundaries)

RECOGNIZING UNSOLVABLE SITUATIONS:
If user's distress stems from another person's addiction, abuse, or uncontrollable behavior:
- Acknowledge the user CANNOT change the other person
- Focus on user's choices: stay/leave, set boundaries/enable, seek support/isolate
- DO NOT suggest "communicate better" or "understand their perspective" - they've tried
- DO recommend external support (Al-Anon for addiction, therapy for user)
- Address the USER's agency and choices, not fixing the other person
- Keep recommendations psychological, not legal/logistical

SAFETY-SENSITIVE SITUATIONS:
If user indicates fear of partner's physical reaction OR pattern suggests domestic violence risk:
- DO NOT recommend direct confrontation or "saying no" statements
- INSTEAD recommend: seeking individual therapy, joining support group, building support network
- Include safety_note in RC-2 with domestic violence hotline
- Keep focus on USER's psychological patterns and agency, not escape planning

THIRD-PARTY PROFESSIONAL REFERRAL:
If situation involves child, partner, or family member with clear mental health/addiction issue:
- Add brief note at END of strategy: "Consider: [child psychologist / addiction specialist] for [person]"
- Keep it SHORT (under 10 words)
- Examples: "Consider: child psychologist for son" / "Consider: addiction specialist for partner"
- This is SECONDARY to user's own psychological work

IMMEDIATE TECHNIQUE REQUIREMENTS:
- Exactly 3 steps (not 4, not 2)
- Executable in 60-90 seconds
- No external resources required
- Targets the core defense mechanism
- Concrete and specific
- MUST match situation type:
  * Anxiety → grounding/breathing
  * Codependency/enabling → boundary-setting micro-action (ONLY if no violence risk)
  * Relationship conflict → specific self-reflection question
  * Learned helplessness → smallest controllable action
  * Permissive parenting → one firm statement to practice
  * Fear of violence → grounding + self-awareness (no confrontation)
- Must give IMMEDIATE sense of agency, not just emotion regulation

IMMEDIATE TECHNIQUE LANGUAGE RULES:
- Use ONLY simple imperative verbs: sit, stand, breathe, write, call, say, go, stop, count, drink, walk
- NO abstract verbs: notice, explore, develop, process, investigate, examine, cultivate
- NO therapeutic jargon: metacognitive, mindfulness, awareness (use simple equivalents)
- Maximum 5-7 words per step
- Physical actions before mental actions: "Sit down. Breathe slowly." comes before "Name your feeling."
- Grade 6 reading level (age 11-12) - simple, direct language
- Technique name must be simple: "Pause 60 seconds" not "Metacognitive awareness practice"
- Example must show exact actions: "Sit on floor. Count to 10." not "Find a quiet space and center yourself."

STRATEGY REQUIREMENTS:
- Exactly 3 actions
- Each action MUST be:
  * SPECIFIC: Include exact words to say or exact steps to take
  * TIMEBOUND: Today, tomorrow, this week, next 3 days
  * ACTIONABLE: Does NOT require others to change first
  * USER-FOCUSED: Addresses USER's agency and choices
  * PROGRESSIVE: Break multi-step actions into "today do X, tomorrow do Y"
  * SAFE: If violence risk detected, avoid confrontational actions
  * PSYCHOLOGICAL: Focus on patterns, feelings, choices - not legal/logistical advice
- Each action uses SIMPLE mechanism labels:
  * Good: "Self-support (CBT)" / "Boundary-setting (DBT)" / "Value work (ACT)"
  * Bad: "Cognitive restructuring with distress tolerance modulation"
- Each action has measurable success criterion
- Progressive difficulty (easy to hard)

SAFETY NOTE FIELD:
- Include safety_note in RC-2 if:
  * User selected fear-of-reaction option
  * Strategy involves boundary-setting with potentially volatile person
  * Suicidal ideation mentioned (not immediate)
- Safety note format (SHORT, under 20 words):
  * "⚠️ If partner has been physically violent: domestic violence hotline [number]"
  * "⚠️ If suicidal thoughts intensify: crisis line [number]"
- If no safety concern: "safety_note": null

ACTION SPECIFICITY RULES:
- Use concrete verbs (write, say, go to, ask yourself, call, find, count, list)
- Avoid abstract verbs (explore, develop, work on, improve, understand, cultivate)
- Include exact wording WITHOUT nested quotes: say to yourself: I am okay (not: say "I am okay")
- Specify quantity (write 3 things, attend 1 meeting, say this once)
- Break complex actions: "Today find phone numbers. Tomorrow call."
- Avoid therapeutic jargon - use plain human language
- NEVER use quotes inside action text - rephrase instead

TONE REQUIREMENTS:
- Direct, clear language - not therapeutic jargon
- Acknowledge user's real constraints
- No false hope or platitudes
- Respect user's intelligence and life experience
- Focus on what user CAN control
- When safety is concern - prioritize user's psychological wellbeing
- In immediate technique: commanding but supportive tone ("Do this." not "You might consider...")

DECISION POINT FORMAT:
After S3, offer choice with MINIMAL text:
- Question text: ONLY "Analysis complete." or equivalent in user language
- Options: ["Deep analysis", "Show result"] or equivalent
- Do NOT add explanatory text like "Would you like to..." or "Choose next step"

DEEP ANALYSIS STRICT RULE:
If user chooses deep analysis (option 1 at decision point):
- Ask S4, wait for answer
- Ask S5, wait for answer
- Ask S6, wait for answer
- ONLY THEN provide RC-2
- Count: decision → S4 → S5 → S6 → RC-2 (3 questions total after decision)
- Do NOT skip S6 and jump to RC-2 after S5

Remember: You are a triage tool, not a therapist. Your goal is rapid pattern identification and SAFE, ACTIONABLE psychological intervention. Physical safety takes priority. Every strategy action must be something the user can DO this week. Immediate technique must be SIMPLE enough for someone in acute stress to follow without thinking.
